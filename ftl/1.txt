CREATE TABLE `credit_order` (
  `order_id` varchar(50) NOT NULL COMMENT '订单号',  有索引
  `mobile` varchar(50) NOT NULL COMMENT '手机号',
  `ext_order_id` varchar(100) DEFAULT NULL COMMENT '外部订单编号',
  `channel_id` varchar(200) DEFAULT NULL COMMENT '渠道id',
  `order_mall_id` int(11) DEFAULT NULL COMMENT '订单所属商城(天猫:100，迪信通:200)',
  `order_gen_scene` int(11) DEFAULT NULL COMMENT '订单产生场景(全款购买:100，完善资料购买:200)',
  `order_total_amount` varchar(100) DEFAULT NULL COMMENT '订单总金额(元)',
  `order_real_amount` varchar(100) DEFAULT NULL COMMENT '实际订单金额(元)',
  `fin_apply_pay_amount` varchar(100) DEFAULT NULL COMMENT '使用的授信额度(元)',
  `cash_amount` varchar(100) DEFAULT NULL COMMENT '现金(元)',
  `item_id` varchar(100) DEFAULT NULL COMMENT 'item_id',
  `sku_id` varchar(100) DEFAULT NULL COMMENT 'sku_id',
  `ext_user_nick` varchar(100) DEFAULT NULL COMMENT '天猫用户id',
  `ali_seller_nick` varchar(100) DEFAULT NULL COMMENT '卖家nick',
  `express_code` varchar(50) DEFAULT NULL COMMENT '物流单号',
  `express_name` varchar(50) DEFAULT NULL COMMENT '物流公司',
  `ext_order_time` datetime DEFAULT NULL COMMENT '天猫订单时间',
  `fin_apply_pay_id` varchar(100) DEFAULT NULL COMMENT '额度支用标识',
  `installment_id` varchar(100) DEFAULT NULL COMMENT '分期规则标识',
  `package_contract` int(11) DEFAULT '0' COMMENT '合约套餐:1,合约套餐,0,非合约套餐',
  `package_type` int(11) DEFAULT '0' COMMENT '办理套餐的类型code,0:初始化,1, 4G飞享套餐,2, 4G飞享流量王,3, 4G新飞享套餐',
  `monthly_fee` int(11) DEFAULT '0' COMMENT '月消',
  `contract_period` int(11) DEFAULT '0' COMMENT '合约期数',
  `package_name` varchar(100) DEFAULT NULL COMMENT '套餐名称',
  `package_details` varchar(300) DEFAULT NULL COMMENT '套餐详细',
  `pck_to_mobile` varchar(100) DEFAULT NULL COMMENT '办理套餐手机号',
  `pck_to_name` varchar(100) DEFAULT NULL COMMENT '办理套餐姓名',
  `pck_to_idcard_num` varchar(100) DEFAULT NULL COMMENT '办理套餐身份在',
  `receiver_name` varchar(100) DEFAULT NULL COMMENT '收货人姓名',
  `receiver_mobile` varchar(100) DEFAULT NULL COMMENT '收货人电话',
  `receiver_address` varchar(100) DEFAULT NULL COMMENT '收货人地址',
  `receiver_province` varchar(100) DEFAULT NULL COMMENT '收货人省',
  `receiver_city` varchar(100) DEFAULT NULL COMMENT '收货人市',
  `receiver_area` varchar(100) DEFAULT NULL COMMENT '收货人区',
  `trade_lng` varchar(100) DEFAULT NULL COMMENT '支用申请授信经度',
  `trade_lat` varchar(100) DEFAULT NULL COMMENT '支用申请授信纬度',
  `trade_province` varchar(50) DEFAULT NULL COMMENT '支用时点LBS地址省',
  `trade_city` varchar(50) DEFAULT NULL COMMENT '支用时点LBS地址市',
  `trade_area` varchar(50) DEFAULT NULL COMMENT '支用时点LBS地址区',
  `trade_ip` varchar(50) DEFAULT NULL COMMENT '支用时点LBS地址区',
  `order_status` int(11) DEFAULT '0' COMMENT '订单状态',
  `apply_pay_fail_code` int(11) DEFAULT '0' COMMENT '订单支用失败原因code,0:初始化,2:金融授信失败,3:金融授信额度小于商品价格,4:金融支用失败',
  `apply_pay_fail_line` varchar(100) DEFAULT '0' COMMENT '订单支用失败原因code,0:如apply_pay_fail_code=3，则记录当时接口返回的可用额度',
  `modify_time` datetime DEFAULT NULL COMMENT '修改时间',
  `create_time` datetime DEFAULT NULL COMMENT '创建时间',
  PRIMARY KEY (`order_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='授信额度使用订单'
